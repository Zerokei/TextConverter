    from pwn import *
p=remote('10.214.160.13',11002)
p.recvlines(7)
for i in range(10):
    s=p.recvuntil('=')
    s=bytes.decode(s)
    if i == 0:
        s=s[:-2]
    else:
        s=s[15:-2]
    sum=eval(s) 
    print(s)
    print(sum)
    sum = str.encode(str(sum))
    p.sendline(sum)
s=p.recvlines(4)
print(s)

    md5加密

    import hashlib
def get_md5(s):
    md = hashlib.md5()
    md.update(s.encode('utf-8'))
    return md.hexdigest()
